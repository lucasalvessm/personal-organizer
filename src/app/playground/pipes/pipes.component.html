O que são pipes?

Pipes são expressões no template para transformação de dados

{{ dateAsString | date : "MM dd, yy, h:mm:ss a z" }}

{{ date2 | date }}

{{ date2 | date : "short" }}

{{ date2 | date : "MM dd, yy, h:mm:ss a z" | lowercase }}

{{ jsonObject | json }}

{{ frase }}

{{ frase | lowercase }}

{{ saldo | currency : "BRL" }}

{{ frase | pipeLegal : "teste" : "asdfds" }}
